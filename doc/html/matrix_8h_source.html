132  std::istringstream stream(str);
133  std::vector<std::string> split(std::istream_iterator<std::string>{stream},
134  std::istream_iterator<std::string>());
135  std::string dimstring = split[0];
136  unsigned open = dimstring.find('[');
137  unsigned comma = dimstring.find(',');
138  unsigned close = dimstring.find(']');
139 
140  size_type m = std::stoi(dimstring.substr(open+1, comma - open));
141  size_type n = std::stoi(dimstring.substr(comma+1, close - comma));
142 
143  unsigned dim = static_cast<unsigned>(m*n);
144  assert(split.size() == dim + 1);
145 
146  Matrix<coeff_t> matrix(m,n);
147  for (auto i : range<size_type>(m))
148  for (auto j : range<size_type>(n))
149  {
150  std::istringstream is(split[i*m + j + 1]);
151  is >> matrix(i,j);
152  }
153  return matrix;
154  }
